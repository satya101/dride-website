Dride-core is our road vision engine. This model is responsible for analyzing the road and cars around it and gives you access to this information in an easy to use API.

var core = require('dride-core')




Return the road area
core.getRoad().then(function(res){ console.log(res) //{"topLeft": "40.556" ,"topRight": "48.441","bottomLeft": "251.11","bottomLeft": "1.44} })
Represent a line from the left corner of the object to the right
core.getFrontCar().then(function(res){ console.log(res) //{"left": {"x": 80.156" : "y": 4.48"},"right": {"x": "71.18","y": "49.41}} })
A number between 0-360 that represent the road angle.
core.roadAngle().then(function(res){ console.log(res) //{"angle": "45.5"} })
A number between 0-250 that represent the current speed.
core.getSpeed().then(function(res){ console.log(res) //{"speed": "90", "unit": "mph"} })
A point that represent the current latitude and longitude.
core.getCurrenPosition().then(function(res){ console.log(res) //{"lat": "34.54454", "lon": "30.488474"} })
